练习 3：循环队列四个操作

#define MAXQSIZE 100
typedef struct {
    int data[MAXQSIZE];
    int front;
    int rear;
} SqQueue;

int enQueue(SqQueue *q, int x)
{
    if ((q->rear + 1) % MAXQSIZE == q->front)
    {
        return 0;
    }

    else
    {
        q->data[q->rear] = x;
        q->rear = (q->rear + 1) % MAXQSIZE;          // rear 后移（取余）
        return 1;
    }
}

int deQueue(SqQueue *q, int *x)
{
    if (q->rear == q->front)
    {
        return 0;
    }

    else
    {
        *x = q->data[q->front];
        q->front = (q->front + 1) % MAXQSIZE;
        return 1;
    }
}
